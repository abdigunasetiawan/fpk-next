"use client";

import { useEffect } from "react";
import { Swiper, SwiperSlide } from "swiper/react";
import { Swiper as SwiperClass } from "swiper";
import { Navigation, Pagination, Autoplay } from "swiper/modules";
import events from "@/json/events";
import EventCard from "./EventCard";

import "swiper/css";
import "swiper/css/navigation";
import "swiper/css/pagination";

interface EventsSwiperProps {
  prevRef: React.RefObject<HTMLButtonElement | null>;
  nextRef: React.RefObject<HTMLButtonElement | null>;
}

const EventsSwiper: React.FC<EventsSwiperProps> = ({ prevRef, nextRef }) => {
  useEffect(() => {
    // hanya debug log kalau ref belum siap
    if (!prevRef.current || !nextRef.current) {
      console.warn("Nav refs belum siap");
    }
  }, [prevRef, nextRef]);

  return (
    <Swiper
      className="swiper-events"
      modules={[Navigation, Pagination, Autoplay]}
      loop
      autoplay={{ delay: 3000, disableOnInteraction: false }}
      slidesPerView={1}
      spaceBetween={16}
      pagination={{
        el: ".swiper-pagination-events",
        clickable: true,
      }}
      breakpoints={{
        768: { slidesPerView: 2 },
        1024: { slidesPerView: 3 },
      }}
      // âœ… onSwiper memberikan akses langsung ke instance swiper
      onSwiper={(swiper: SwiperClass) => {
        if (!prevRef.current || !nextRef.current) return;

        if (
          swiper.params.navigation &&
          typeof swiper.params.navigation !== "boolean"
        ) {
          swiper.params.navigation.prevEl = prevRef.current;
          swiper.params.navigation.nextEl = nextRef.current;
          swiper.navigation.destroy();
          swiper.navigation.init();
          swiper.navigation.update();
        }
      }}
    >
      {events.map((event) => (
        <SwiperSlide key={event.id}>
          <EventCard event={event} />
        </SwiperSlide>
      ))}

      <div className="swiper-pagination-events swiper-pagination" />
    </Swiper>
  );
};

export default EventsSwiper;
